int power[]={2,3,4,5,6};int gnd[]={7,8,9,10};void setup(){ for(int i=0; i<5; i++){  pinMode(power[i],OUTPUT); }   for(int i=0; i<4; i++){   pinMode(gnd[i],OUTPUT);   digitalWrite(gnd[i],HIGH);  }}void loop(){ for(int i=0; i<4;i++){    for(int j=0; j<5;j++){    digitalWrite(gnd[i],LOW);    digitalWrite(power[j],HIGH);    delay(100);     digitalWrite(gnd[i],HIGH);    digitalWrite(power[j],LOW);        delay(100);    } }}
