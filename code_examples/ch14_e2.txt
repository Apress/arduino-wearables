int speaker=1;int skinSensor=1;  void setup() {                 pinMode(speaker, OUTPUT);    }void loop() {  int skinValue=analogRead(skinSensor);   if(skinValue<200){     for(int i =0; i<30; i++){      digitalWrite(speaker, HIGH);       delayMicroseconds(skinValue *3);                  digitalWrite(speaker, LOW);          delayMicroseconds(skinValue *3);    }  }  }
