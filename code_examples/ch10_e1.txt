int gndLine1 = 2; int gndLine2 = 3; int powerLine1 = 4; int powerLine2 = 5; void setup(){pinMode(powerLine1,OUTPUT);   pinMode(powerLine2,OUTPUT); pinMode(gndLine1,OUTPUT); pinMode(gndLine2,OUTPUT); digitalWrite(gndLine1,HIGH); digitalWrite(gndLine1,HIGH);}void loop(){activateLine(gndLine1,powerLine1);  delay(500);activateLine(gndLine1,powerLine2);delay(500);activateLine(gndLine2,powerLine1);delay(500);activateLine(gndLine2,powerLine2);delay(500);}void activateLine(int gnd, int power){ digitalWrite(gndLine1,HIGH); digitalWrite(gndLine2,HIGH); digitalWrite(powerLine1,LOW);    digitalWrite(powerLine2,LOW);  digitalWrite(gnd, LOW);   digitalWrite(power, HIGH);}
