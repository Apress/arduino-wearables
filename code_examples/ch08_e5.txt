int zipperValue=0;int zipperPin=0;int piezo=9; int buttonPin=11;int buttonState=0; int keyCode=0; void setup(){  pinMode(piezo, OUTPUT);  pinMode(buttonPin, INPUT);  digitalWrite(buttonPin,HIGH); }void loop(){  noTone(9);   zipperValue=analogRead(zipperPin);  if(zipperValue >= 10 && zipperValue <= 50 && keyCode==0){    beep();     keyCode++;   }  if(zipperValue >= 50 && zipperValue <= 100 && keyCode==1){    beep();    keyCode++;  }  if(zipperValue >= 100 && zipperValue <= 150 && keyCode==2){    beep();     keyCode++;  }  if(zipperValue >= 150 && zipperValue <= 200 && keyCode==3){    beep();     keyCode++;  }  if(zipperValue >= 200 && zipperValue <= 250 && keyCode==4){    beep();     keyCode++;   }  if(digitalRead(buttonPin)==HIGH){    if(keyCode != 5){      soundAlarm(500,1000);    }    while(digitalRead(buttonPin)==HIGH){};    keyCode=0;  }  delay(300);}void soundAlarm(int delay1, int delay2){ while(digitalRead(buttonPin)==HIGH){  for(int i=0; i<500; i++){   tone(9,delay1,100);  }  for(int i=0; i<500; i++){   tone(9,delay2,100);  }  }}  void beep(){ for(int i=0; i<50; i++){  tone(9,800,100); }}
