int flower=10;int ldr=5; long heatLimit; boolean onOk = true;  void setup(){ pinMode(flower,OUTPUT);  Serial.begin(9600);  heatLimit = 10000+millis(); }void loop(){ long timeOn=millis();  digitalWrite(flower,LOW); int ldrValue=analogRead(ldr); while(ldrValue<500 && onOk){    ldrValue=analogRead(ldr);    digitalWrite(flower,HIGH);   if(millis()>heatLimit){     onOk=false;    } } if(!onOk){   digitalWrite(flower,LOW);   delay(60000);   heatLimit = 10000+millis();    onOk=true;  } }
