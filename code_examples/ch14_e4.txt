int pressureSensor = 1;  int eye1= 0;int eye2= 1;int angryTime=10;int coolDownTime; void setup() {  pinMode(eye1,OUTPUT);   pinMode(eye2,OUTPUT); }void loop() {  int time=millis();   if(angryTime==1 && time> coolDownTime){    angryTime=10;   }  while(analogRead(pressureSensor)<400){     coolDownTime=millis()+5000;     if(angryTime>1){      for(int i =0; i<255; i++){         analogWrite(eye1,i);        analogWrite(eye2,i);         delay(angryTime);      }      for(int i =255; i>0; i--){        analogWrite(eye1,i);        analogWrite(eye2,i);        delay(angryTime);      }    }      angryTime-=2;    if(angryTime<=1){      angryTime=1;       digitalWrite(eye1,HIGH);      digitalWrite(eye2,LOW);      delay(100);      digitalWrite(eye1,LOW);      digitalWrite(eye2,HIGH);       delay(100);    }  }  digitalWrite(eye1,LOW);  digitalWrite(eye2,LOW);}
