int zipperValue=0;int zipperPin=0;int piezo=9; void setup(){ Serial.begin(9600);  pinMode(piezo,OUTPUT); }void loop(){ zipperValue=analogRead(zipperPin); if(zipperValue<1000){   int newZipperValue=map(zipperValue, 0, 300, 500, 2000);    digitalWrite(piezo,HIGH);   delayMicroseconds(newZipperValue);    digitalWrite(piezo,LOW);   delayMicroseconds(newZipperValue);  }}
