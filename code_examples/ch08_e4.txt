int zipperValue=0;int zipperPin=0;int piezo=9; int buttonPin=11;int buttonState=0; void setup(){ noTone(9);  Serial.begin(9600);  pinMode(piezo,OUTPUT);  pinMode(buttonPin,INPUT);}void loop(){ buttonState=digitalRead(buttonPin);   if(buttonState==LOW){  int zipperValue=analogRead(zipperPin);     int newZipperValue=map(zipperValue, 0, 300, 0, 2000);   if(newZipperValue > 120 && newZipperValue < 140){    noTone(9);   }else{    soundAlarm(500,1000);   }   }}void soundAlarm(int delay1, int delay2){  while(buttonState==LOW){    for(int i=0; i<500; i++){      tone(9,delay1,100);    }    for(int i=0; i<500; i++){      tone(9,delay2,100);    }    buttonState=digitalRead(buttonPin);   }}
