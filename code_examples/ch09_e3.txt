int piezo = 7;int button[]={9,10,11,12,13,14,15,16};int buttonPress[] = {0,0,0,0,0,0,0,0,0};int tonePitch[] = {0,0,0,0,0,0,0,0};int mic=3; long timer=0; int micMin=340;int micMax=350;void setup(){ for(int i=0; i<8; i++){   pinMode(button[i], INPUT);   digitalWrite(button[i],HIGH);  }  pinMode(piezo,OUTPUT); }void loop(){    playBeat(buttonPress[0], tonePitch[0],20);     playBeat(buttonPress[1], tonePitch[1],20);     playBeat(buttonPress[2], tonePitch[2],20);     playBeat(buttonPress[3], tonePitch[3],20);     playBeat(buttonPress[4], tonePitch[4],20);     playBeat(buttonPress[5], tonePitch[5],20);     playBeat(buttonPress[6], tonePitch[6],20);     playBeat(buttonPress[7], tonePitch[7],20);  }void playBeat(int times, int delayTime, int duration){  timer=millis();  voiceActive();  for(int j=0; j<times; j++){    for(int k=0; k<duration; k++){   digitalWrite(piezo,HIGH);   delayMicroseconds(delayTime);    digitalWrite(piezo,LOW);   delayMicroseconds(delayTime);    voiceActive();   }   delay(200);   }        if(times==0){         delay(200);        }}void voiceActive(){    for(int k=0; k<8; k++){    if(digitalRead(button[k])==LOW){      buttonPress[k]=0;;      while(digitalRead(button[k])==LOW){       if(analogRead(mic)>micMax ||analogRead(mic)<micMin ){         buttonPress[k]++;         while(analogRead(mic)>350 ||analogRead(mic)<340 ){             }        }      long newTime=millis()-timer;        tonePitch[k]=newTime*2.5;       }    }  } }
