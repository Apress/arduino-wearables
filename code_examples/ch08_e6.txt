int delayTimes[] ={1047, 1175, 1319, 1397, 1568, 1760, 1976}; char toneNames[] = { 'C', 'D', 'E', 'F', 'G', 'B', 'A',}; char melody[] = { 'E', 'F', 'E', 'G', 'D', 'A', 'B', 'A', 'E', 'E', 'A', 'B', 'A', 'A',}; int piezo=9; void setup(){  pinMode(piezo, OUTPUT); }void loop(){  playMelody();}void playMelody(){ for(int i=0; i<sizeof(melody); i++){   for(int j=0; j<sizeof(toneNames); j++){     if(melody[i]==toneNames[j]){      for(int k=0; k<50; k++){        digitalWrite(piezo, HIGH);       delayMicroseconds(delayTimes[j]);       digitalWrite(piezo, LOW);       delayMicroseconds(delayTimes[j]);      }     }    } }}
