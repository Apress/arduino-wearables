int power[]={2,3,4,5,6};int gnd[]={7,8,9,10};void setup(){ for(int i=0; i<5; i++){  pinMode(power[i],OUTPUT); }   for(int i=0; i<4; i++){       pinMode(gnd[i],OUTPUT);   digitalWrite(gnd[i],HIGH);  }}void loop(){   setLed(2,500);    setLed(3,500);    setLed(12,500);    setLed(20,500); }void setLed(int led, int time){  switch (led){    case 1:    flashLed(gnd[0],power[0],time);    break;    case 2:    flashLed(gnd[0],power[1],time);    break;    case 3:     flashLed(gnd[0],power[2],time);    break;     case 4:    flashLed(gnd[0],power[3],time);    break;    case 5:     flashLed(gnd[0],power[4],time);    break;     case 6:    flashLed(gnd[1],power[0],time);    break;    case 7:     flashLed(gnd[1],power[1],time);    break;     case 8:    flashLed(gnd[1],power[2],time);    break;        case 9:     flashLed(gnd[1],power[3],time);    break;     case 10:    flashLed(gnd[1],power[4],time);       break;    case 11:    flashLed(gnd[2],power[0],time);     break;     case 12:    flashLed(gnd[2],power[1],time);    break;    case 13:    flashLed(gnd[2],power[2],time);     break;     case 14:    flashLed(gnd[2],power[3],time);      break;    case 15:     flashLed(gnd[2],power[4],time);           break;     case 16:    digitalWrite(gnd[3],LOW);     digitalWrite(power[0],HIGH);    delay(time);    digitalWrite(gnd[3],HIGH);     digitalWrite(power[0],LOW);    delay(time);     break;    case 17:    flashLed(gnd[3],power[1],time);     break;     case 18:    flashLed(gnd[3],power[2],time);    break;     case 19:    flashLed(gnd[3],power[3],time);    break;     case 20:    flashLed(gnd[3],power[4],time);    break;      }}void flashLed(int ground, int power, int delayTime){      digitalWrite(ground,LOW);     digitalWrite(power,HIGH);    delay(delayTime);    digitalWrite(ground,HIGH);     digitalWrite(power,LOW);    delay(delayTime);}
