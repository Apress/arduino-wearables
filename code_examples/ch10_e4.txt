int power[]={2,3,4,5,6};int gnd[]={7,8,9,10};int sensor1=3;int sensor2=2;void setup(){ for(int i=0; i<5; i++){  pinMode(power[i],OUTPUT); }   for(int i=0; i<4; i++){   pinMode(gnd[i],OUTPUT);   digitalWrite(gnd[i],HIGH);  } randomSeed(analogRead(0)); }void loop(){  int rainSense1 = analogRead(sensor1);  int rainSense2 = analogRead(sensor2);  if(rainSense1 < 10 || rainSense2 < 10){   randomRain();   }       }void randomRain(){  int gndRandom = random(0,5);  int powerRandom = random(0,6);  digitalWrite(gnd[gndRandom],LOW);  digitalWrite(power[powerRandom],HIGH);  delay(30);  digitalWrite(gnd[gndRandom],HIGH);  digitalWrite(power[powerRandom],LOW);     delay(30);  }
