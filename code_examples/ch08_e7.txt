int delayTimes[] ={1911 , 1500 , 1400 , 1300 , 1211 , 1054, 1}; char toneNames[] = { 'C', 'D', 'E', 'F', 'G', 'B', 'A', 'P'}; char melody[] = { 'D', 'P', 'B', 'D', 'A', 'D', 'P', 'G', 'A', 'D', 'P', 'A', 'D', 'P'}; int duration[] = {8,8,1,8,8,4,8,8,4,4,8}; int piezo=9; int zipperPin=0; int zipperValue=0;int bpm = 0;int buttonPin=11; int buttonState=0;void setup(){  pinMode(piezo, OUTPUT);   pinMode(buttonPin, INPUT);   digitalWrite(buttonPin,HIGH);}void loop(){  buttonState = digitalRead(buttonPin);  if(buttonState == LOW){      playMelody();    }  }void playMelody(){  zipperValue=analogRead(zipperPin);     for(int i=0; i<sizeof(melody); i++){    for(int j=0; j<sizeof(toneNames); j++){      if(melody[i]==toneNames[j]){        for(int k=0; k<map(analogRead(zipperPin), 0 , 1023, 0 ,     2000)/duration[j]; k++){           digitalWrite(piezo, HIGH);          delayMicroseconds(delayTimes[j]);          digitalWrite(piezo, LOW);          delayMicroseconds(delayTimes[j]);        }        delay(duration[j]*1.40);       }     }  }}
