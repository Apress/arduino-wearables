#define A  2#define B  3#define C  4#define D  5void setup(){  //Declare all pins as OUTPUT  pinMode(A, OUTPUT);  pinMode(B, OUTPUT);  pinMode(C, OUTPUT);  pinMode(D, OUTPUT);}void loop(){   digitalWrite(A, HIGH);   delay(2000);    digitalWrite(A, LOW);   delay(500);    digitalWrite(B, HIGH);   delay(2000);    digitalWrite(B, LOW);   delay(500);    digitalWrite(C, HIGH);   delay(2000);    digitalWrite(C, LOW);   delay(500);    digitalWrite(D, HIGH);   delay(2000);    digitalWrite(D, LOW);   delay(500);}
